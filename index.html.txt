<script>

const defaultConfig={
version:1,
playmats:{
white:"https://example.com/playmat1.jpg"
},
sleeves:{
waves:"https://example.com/sleeve1.png"
}
}

document.getElementById("config").value=
JSON.stringify(defaultConfig,null,2)

function bumpVersion(){

let cfg=JSON.parse(document.getElementById("config").value)

cfg.version++

document.getElementById("config").value=
JSON.stringify(cfg,null,2)

}

function downloadConfig(){

let text=document.getElementById("config").value

let blob=new Blob([text],{type:"application/json"})

let a=document.createElement("a")

a.href=URL.createObjectURL(blob)

a.download="config.json"

a.click()

}

function generateScript(){

let url=document.getElementById("configUrl").value

let script=`// ==UserScript==
// @name EDHPlay Custom Assets
// @match https://edhplay.com/*
// @grant GM_xmlhttpRequest
// @connect *
// ==/UserScript==

(function(){

const CONFIG_URL="${url}"

let config={playmats:{},sleeves:{}}

function loadConfig(){

GM_xmlhttpRequest({
method:"GET",
url:CONFIG_URL+"?"+Date.now(),
onload:function(r){

config=JSON.parse(r.responseText)

apply()

}

})

}

function apply(){

document.querySelectorAll("img").forEach(img=>{

let src=img.src

for(let k in config.sleeves){

if(src.includes("/sleeves/"+k)){

img.src=config.sleeves[k]

}

}

})

document.querySelectorAll('[data-zone-name="battlefield"]').forEach(zone=>{

let inner=zone.firstElementChild

let bg=getComputedStyle(zone).backgroundImage

for(let k in config.playmats){

if(bg.includes("/playmats/"+k)){

zone.style.backgroundImage="url("+config.playmats[k]+")"

if(inner)inner.style.backgroundImage="url("+config.playmats[k]+")"

}

}

})

}

setInterval(loadConfig,30000)

loadConfig()

})()
`

document.getElementById("script").value=script

}

function copyScript(){

navigator.clipboard.writeText(
document.getElementById("script").value
)

}

function downloadScript(){

let text=document.getElementById("script").value

let blob=new Blob([text],{type:"text/javascript"})

let a=document.createElement("a")

a.href=URL.createObjectURL(blob)

a.download="edhplay.user.js"

a.click()

}
